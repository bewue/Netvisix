libtins MinGW static build:
(http://libtins.github.io/download/#compiling-windows)
cmake ../ -DLIBTINS_BUILD_SHARED=0 -DLIBTINS_ENABLE_CXX11=0 -DLIBTINS_ENABLE_WPA2=0 -DPCAP_ROOT_DIR=<npcap-sdk-dir> -G "MinGW Makefiles" -DLIBTINS_ENABLE_DOT11=0
cmake --build .
